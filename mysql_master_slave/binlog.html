# 主配置

  • 开启binlog
  • 配置SERVER_ID
  • 创建授权账号
  • 查看主信息

# master - 开启binlog

show variables like '%log_bin%'

[mysqld]
log-bin=mysql-bin

# 从配置

  • 配置SERVER_ID
  • 配置主的信息
  • 启动从服务
  • 查看主从状态

# slave - 配置SERVER_ID

  • servier-id 需要唯一，且不能和主重复
vim /etc/my.cnf

[mysqld]
server-id = 1

4

###slave - 配置主信息

>mysql -uroot -p

change master to
master_host='xxx.xxx.xxx.xxx',
master_user='slave',
master_password='#f1d54mi',
master_log_file='mysql-bin.000006',
master_log_pos=154;

# slave - 查看主从状态

show slave status\G

    • Slave_IO_Running : Yes
    • Slave_SQL_Running : Yes

# slave - 启动从服务
